<script>
$(document).ready(function() {
let fishslideIndex = 1;
fishshowSlides(fishslideIndex);

function fishplusSlides(n) {
  fishshowSlides(fishslideIndex += n);
}

function fishcurrentSlide(n) {
  fishshowSlides(fishslideIndex = n);
}

function fishshowSlides(n) {
  let i;
  let fishslides = document.getElementsByClassName("fishSlides");
  if (n > fishslides.length) {fishslideIndex = 1}    
  if (n < 1) {fishslideIndex = fishslides.length}
  for (i = 0; i < fishslides.length; i++) {
    fishslides[i].style.display = "none";  
  }
  fishslides[fishslideIndex-1].style.display = "block";  
}
});
</script>
