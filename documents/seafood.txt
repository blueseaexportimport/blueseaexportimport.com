<script>
let slideIndex = 1;
showSlides(slideIndex);

function plusSlides(n) {
  showSlides(slideIndex += n);
}

function currentSlide(n) {
  showSlides(slideIndex = n);
}

function showSlides(n) {
  let i;
  let slidesseafood = document.getElementsByClassName("seafoodSlides");
  if (n > slidesseafood.length) {slideIndex = 1}    
  if (n < 1) {slideIndex = slidesseafood.length}
  for (i = 0; i < slidesseafood.length; i++) {
    slidesseafood[i].style.display = "none";  
  }
  slidesseafood[slideIndex-1].style.display = "block";  
}
</script>
