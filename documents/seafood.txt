<script>
let seafoodslideIndex = 1;
seafoodshowSlides(seafoodslideIndex);

function seafoodplusSlides(n) {
  seafoodshowSlides(seafoodslideIndex += n);
}

function currentSlide(n) {
  seafoodshowSlides(seafoodslideIndex = n);
}

function seafoodshowSlides(n) {
  let i;
  let seafoodslides = document.getElementsByClassName("seafoodSlides");
  if (n > seafoodslides.length) {seafoodslideIndex = 1}    
  if (n < 1) {seafoodslideIndex = seafoodslides.length}
  for (i = 0; i < seafoodslides.length; i++) {
    seafoodslides[i].style.display = "none";  
  }
  seafoodslides[seafoodslideIndex-1].style.display = "block";  
}
</script>
